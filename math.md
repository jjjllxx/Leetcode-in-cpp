## Modular Arithmetic
### basic
1. (a + b) % p = (a % p + b % p) % p 
2. (a - b) % p = (a % p - b % p) % p
3. (a * b) % p = (a % p * b % p) % p
4. a<sup>b</sup> % p = ((a % p)<sup>b</sup>) % p
5. $Σ_n^1x$ % p = ($Σ_n^1x$ % p) % p

### Associative Property
1. ((a + b) % p + c) % p = (a + (b + c) % p) % p  
2. ((a * b) % p * c) % p = (a * (b * c) % p) % p

### Commutative Property
1. (a + b) % p = (b + a) % p
2. (a * b) % p = (b * a) % p

### Distributive Property
1. (a + b) % p=(a % p + b % p) % p
2. ((a + b) % p * c) % p = ((a * c) % p + (b * c) % p) % p
